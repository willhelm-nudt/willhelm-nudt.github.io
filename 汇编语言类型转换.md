void wtoc(int num,char *str) 
{
    int rem=0;
    char c='\0';

    Stack s;
    Init(&s);
    Push(s,'\0');

    while(num!=0)
    {
      rem=num % 10;
      c=rem+'0';
      Push(&s,c);
      num /= 10;
    }
    do{
      c=Pop(&s);
      *str=c;
      ++str;
    }while(c!='\0');
    Destroy(&s);
}
